% ============================================================
% newword.tex
% ------------------------------------------------------------
% Definição do comando \newword utilizado por docs/tese/glossario.tex.
%
% Sintaxe de uso:
%     \newword{termo}{explicação}
%
% Renderização:
%     - Termo em negrito, alinhado à esquerda.
%     - Explicação em texto corrente, com recuo suave à direita.
%     - Separação vertical entre entradas para leitura confortável.
%
% Como incluir na sua dissertação (main.tex):
%     1. No preâmbulo, imediatamente antes do \begin{document}:
%            \input{newword}
%     2. No corpo do documento, onde o glossário deve aparecer:
%            \input{glossario}
%
% O ambiente {glossario} envolve automaticamente a lista de entradas
% definida em glossario.tex, produzindo cabeçalho de seção e entrada
% no sumário. Se preferir renderização diferente (ex.: sem entrada no
% sumário, ou como capítulo em vez de seção), basta ajustar aqui.
% ============================================================

% Requisitos (adicionar ao preâmbulo do main.tex se ainda não estiverem):
%   \usepackage{enumitem}
%   \usepackage{hyperref}   % opcional, para links de cross-reference

% ------------------------------------------------------------
% Comando principal — cada entrada do glossário
% ------------------------------------------------------------
\newcommand{\newword}[2]{%
  \item[\textbf{#1}]\hfill\\%
  \noindent\hspace*{0em}#2%
  \vspace{0.4em}%
}

% ------------------------------------------------------------
% Ambiente de contenção — envolve o corpo do glossário
% ------------------------------------------------------------
\newenvironment{glossario}
{%
  \section*{Glossário}%
  \addcontentsline{toc}{section}{Glossário}%
  \begin{description}[%
      style=nextline,%
      leftmargin=0em,%
      itemsep=0.2em,%
      parsep=0em%
  ]%
}
{%
  \end{description}%
}
