% ============================================================
% 1. Datasets faciais e de tom de pele
% ============================================================

\newword{FairFace}{Dataset facial balanceado publicado por Kärkkäinen e Joo (2021), contendo 108.501 imagens distribuídas aproximadamente igual entre sete categorias raciais (White, Black, Latinx/Hispanic, East Asian, Southeast Asian, Indian e Middle Eastern). Referência canônica para \emph{race classification} multi-classe.}

\newword{RFW}{Racial Faces in the Wild --- dataset introduzido por Wang et al. (2019), balanceado em quatro categorias raciais e organizado em pares para tarefas de verificação um-para-um. Adotado neste trabalho como \emph{benchmark} de transferência \emph{fair} para \emph{face recognition}.}

\newword{BFW}{Balanced Faces in the Wild --- dataset publicado por Robinson et al. (2020), estruturado em oito subgrupos definidos pela interseção de raça e gênero. Alternativa a RFW para avaliação intersectional.}

\newword{Casual Conversations}{Dataset de vídeo introduzido por Hazirbas et al. (2021) com anotação de gênero \emph{self-reported}, idade, escala Fitzpatrick e escala Monk Skin Tone rotuladas por anotadores treinados. A versão v2 (Porgali et al. 2023) cobre sete países.}

\newword{STW}{Skin Tone in the Wild --- dataset publicado por Matias et al. (2026), contendo 42.313 imagens de 3.564 indivíduos com anotação MST em dez pontos. Base de treinamento do SkinToneNet.}

\newword{MST-E}{Monk Skin Tone Examples --- dataset auxiliar de referência estabelecido por Schumann et al. (2023) para calibração e consenso de anotação MST.}

% ============================================================
% 2. Modelos e arquiteturas
% ============================================================

\newword{ConvNeXt-T}{Backbone convolucional moderno proposto por Liu et al. (2022) como sucessor arquitetural da família ResNet. Variante T (\emph{Tiny}) com aproximadamente 28 milhões de parâmetros. Backbone adotado nesta dissertação.}

\newword{ResNet-34}{Rede neural residual profunda com 34 camadas, referência histórica em visão computacional. Baseline canônico originalmente treinado por Kärkkäinen e Joo (2021) sobre o FairFace, produzindo 72\% de acurácia agregada.}

\newword{SkinToneNet}{Classificador de tom de pele em escala Monk Skin Tone publicado por Matias et al. (2026). Vision Transformer Small \emph{fine-tuned} no dataset STW. Adotado como insumo pré-treinado na Etapa 1 do pipeline.}

\newword{FaceScanPaliGemma}{Modelo de classificação racial multi-classe sobre FairFace introduzido por AlDahoul et al. (2024), obtido por \emph{fine-tuning} de PaliGemma reformulando classificação como \emph{Visual Question Answering}. Estado da arte atual, com F1 macro de 75\% e disparidade severa entre classes (Black 90\%, Latinx 60\%).}

\newword{CLIP}{Contrastive Language-Image Pre-training --- modelo de \emph{vision-language} proposto por Radford et al. (2021) que aprende representações conjuntas de imagem e texto por meio de aprendizado contrastivo. Fornece \emph{embeddings} semânticos utilizados como sinal condicionante alternativo na Configuração D do estudo comparativo.}

\newword{PaliGemma}{Modelo de \emph{vision-language} da família Google/Gemma que combina codificador de imagens SigLIP com decodificador de linguagem Gemma-2B. Base do FaceScanPaliGemma.}

\newword{ViT}{Vision Transformer --- família de arquiteturas que aplica o mecanismo \emph{Transformer} diretamente sobre \emph{patches} de imagem. Base do SkinToneNet (variante \emph{Small}).}

% ============================================================
% 3. Escalas de tom de pele
% ============================================================

\newword{Monk Skin Tone (MST)}{Escala de tom de pele em dez pontos desenvolvida em parceria entre Ellis Monk (Harvard) e Google, formalizada por Schumann et al. (2023). Projetada especificamente para auditoria de \emph{fairness} em IA, com maior representatividade fototípica que a escala Fitzpatrick.}

\newword{Fitzpatrick}{Escala de fototipia cutânea em seis tipos desenvolvida por Fitzpatrick (1988) para fototerapia dermatológica. Amplamente adotada em pesquisa de \emph{fairness} apesar de viés caucasiano-cêntrico documentado.}

\newword{ITA}{Individual Typology Angle --- métrica contínua de tom de pele derivada dos canais L$^\ast$, a$^\ast$, b$^\ast$ do espaço de cor CIE. Alternativa contínua às escalas discretas MST e Fitzpatrick.}

% ============================================================
% 4. Mecanismos de conditioning
% ============================================================

\newword{FiLM}{Feature-wise Linear Modulation --- mecanismo arquitetural de \emph{conditioning} proposto por Perez et al. (2018). Aplica transformação afim $\gamma \odot F + \beta$ por canal sobre \emph{feature maps} intermediárias, com $\gamma$ e $\beta$ derivados do sinal condicionante. Adotado como linha principal da dissertação.}

\newword{Gated FiLM}{Variante não-linear do FiLM padrão que aplica \emph{gating} sigmoide sobre as funções de modulação: $F' = \sigma(\gamma_g) \odot \tanh(\gamma_t) \odot F + \beta$. Utilizada na Configuração C do estudo comparativo como \emph{ablation} da linearidade.}

\newword{Conditional Batch Normalization}{Mecanismo de \emph{conditioning} anterior ao FiLM, generalizado por este último. Modula os parâmetros de normalização de batch com base em sinal externo.}

\newword{Cross-attention}{Mecanismo de \emph{attention} entre duas sequências distintas (\emph{query} de uma, \emph{key/value} da outra), típico de \emph{Transformers}. Considerado como alternativa e descartado por superdimensionamento para sinal MST de dez dimensões.}

\newword{AdaIN}{Adaptive Instance Normalization --- técnica de \emph{style transfer} que modula estatísticas de instância. Descartada por incompatibilidade conceitual com \emph{conditioning} demográfico global.}

\newword{SPADE}{Spatially Adaptive Normalization --- técnica de \emph{conditioning} espacial. Descartada por exigir mapa espacial denso como sinal condicionante, incompatível com vetor MST global.}

\newword{HyperNetworks}{Meta-arquitetura em que uma rede neural gera os pesos de outra rede. Descartada por instabilidade de treinamento e sobredimensionamento para o cenário desta dissertação.}

\newword{LoRA}{Low-Rank Adaptation --- técnica de \emph{parameter-efficient fine-tuning} que injeta matrizes de baixo posto nas camadas do \emph{backbone}. Modifica pesos, não realiza \emph{conditioning} de \emph{features} por sinal externo.}

% ============================================================
% 5. Métricas de fairness e estatística
% ============================================================

\newword{F1 macro}{Média aritmética simples do F1 por classe. Métrica de desempenho agregada que trata todas as classes com o mesmo peso, sensível a desempenho ruim em classes minoritárias.}

\newword{Worst-class F1}{F1 da classe com pior desempenho no conjunto de teste. Métrica alinhada ao princípio de \emph{Distributionally Robust Optimization}, penalizando disparidades entre classes.}

\newword{Disparity Ratio (DR)}{Razão entre o F1 da pior classe e o F1 da melhor classe: $DR = \min_c F1_c / \max_c F1_c$. Varia entre 0 e 1; valores próximos de 1 indicam maior equidade entre classes.}

\newword{Equal Opportunity}{Definição formal de \emph{fairness} proposta por Hardt et al. (2016). Exige que a taxa de verdadeiros positivos (TPR) seja igual entre grupos demográficos para o resultado favorável.}

\newword{Equalized Odds}{Definição formal de \emph{fairness} de Hardt et al. (2016) mais restritiva que \emph{Equal Opportunity}: exige igualdade simultânea de TPR e FPR entre grupos.}

\newword{TPR}{True Positive Rate --- taxa de verdadeiros positivos. Fração de instâncias positivas corretamente classificadas como positivas.}

\newword{FPR}{False Positive Rate --- taxa de falsos positivos. Fração de instâncias negativas incorretamente classificadas como positivas.}

\newword{Teorema da Impossibilidade}{Resultado formal de Kleinberg et al. (2017) que estabelece a impossibilidade de satisfazer simultaneamente as definições de \emph{calibration}, \emph{balance for the positive class} e \emph{balance for the negative class} exceto em casos degenerados. Motiva a triangulação de métricas adotada no Capítulo de Metodologia.}

\newword{Fleiss' kappa}{Coeficiente estatístico de concordância entre múltiplos anotadores para variáveis categóricas.}

\newword{Cohen's kappa}{Coeficiente estatístico de concordância entre dois anotadores para variáveis categóricas. Adotado nesta dissertação para a validação humana interna do OE1.}

\newword{Bootstrap}{Técnica não paramétrica de reamostragem por reposição utilizada para estimar intervalos de confiança e distribuições amostrais sem suposição de normalidade.}

\newword{Comparação pareada casada}{Protocolo estatístico em que duas configurações são avaliadas sobre exatamente os mesmos \emph{splits} de dados e as mesmas inicializações, permitindo estatísticas de diferença com maior poder inferencial.}

% ============================================================
% 6. Baselines de mitigação algorítmica
% ============================================================

\newword{FSCL+}{Fair Supervised Contrastive Learning --- extensão de SupCon proposta por Park et al. (2022) que remove informação demográfica das representações contrastivas.}

\newword{Group DRO}{Group Distributionally Robust Optimization --- formalização por Sagawa et al. (2020) que otimiza o pior \emph{worst-case group}, oferecendo garantia de desempenho para o grupo minoritário.}

\newword{FineFACE}{Método de \emph{fairness} facial baseado em \emph{cross-layer mutual attention learning} proposto por Manzoor et al. (2024), atingindo operação Pareto-eficiente em CelebA.}

\newword{Adversarial debiasing}{Família de técnicas que introduz um classificador adversarial para atributos demográficos durante o treinamento, forçando o modelo principal a produzir representações independentes desses atributos. Referência canônica: Zhang et al. (2018).}

\newword{LAFTR}{Learning Adversarially Fair and Transferable Representations --- método de Madras et al. (2018) que prova formalmente a herança de propriedades de \emph{fairness} de representações para tarefas \emph{downstream}.}

\newword{LFR}{Learning Fair Representations --- paradigma inaugurado por Zemel et al. (2013), agraciado com o \emph{Test-of-Time Award} da ICML em 2023.}

% ============================================================
% 7. Conceitos técnicos gerais
% ============================================================

\newword{Face recognition (FR)}{Tarefa de identificar ou verificar identidade de uma pessoa a partir de imagem facial. Inclui subtarefas de verificação (um-para-um) e identificação (um-para-N).}

\newword{Race classification}{Tarefa de classificar imagens faciais em uma dentre várias categorias raciais predefinidas. Objeto central desta dissertação em sua variante \emph{multi-classe} (sete classes do FairFace).}

\newword{VLM}{Vision-Language Model --- família de modelos que aprende representações conjuntas de imagem e texto, como CLIP, PaliGemma e LLaVA. Objeto da Seção~2.3 da Revisão de Literatura.}

\newword{VQA}{Visual Question Answering --- tarefa em que o modelo responde a perguntas em linguagem natural sobre uma imagem. Reformulação empregada por AlDahoul et al. (2024) para \emph{race classification}.}

\newword{Backbone}{Arquitetura de rede neural principal que produz representações intermediárias, sobre a qual são aplicados os \emph{heads} de tarefa específica. Nesta dissertação, o \emph{backbone} adotado é o ConvNeXt-T.}

\newword{Fine-tuning}{Procedimento de continuação de treinamento de um modelo pré-treinado sobre novo dataset ou nova tarefa, tipicamente com \emph{learning rate} reduzido.}

\newword{Pipeline}{Sequência ordenada de estágios de processamento e/ou modelos que, combinados, executam uma tarefa complexa. Nesta dissertação, o pipeline principal é composto por seis etapas (Seção~4.2).}

\newword{Ablation}{Estudo experimental que remove ou substitui sistematicamente componentes de um sistema para isolar sua contribuição individual. Nesta dissertação, a \emph{ablation} arquitetural do Cap.~2 compara quatro configurações de \emph{conditioning}.}

\newword{Conditioning}{Mecanismo pelo qual um modelo neural altera seu comportamento em função de um sinal auxiliar externo. Nesta dissertação, o sinal é o vetor MST 10-dim produzido pelo SkinToneNet, e o mecanismo é FiLM.}

\newword{Feature map}{Representação intermediária tridimensional produzida por camadas convolucionais, com dimensões de largura, altura e canais.}

\newword{Fair transferência}{Propriedade pela qual as garantias de \emph{fairness} obtidas em uma tarefa (\emph{upstream}) preservam-se, ao menos em limite superior, quando o modelo é aplicado a uma tarefa distinta (\emph{downstream}). Fundamentada teoricamente por Madras et al. (2018).}

\newword{Downstream task}{Tarefa aplicada após a \emph{upstream} (tipicamente pré-treinamento ou treinamento inicial). Nesta dissertação, \emph{race classification} é \emph{upstream} e \emph{face recognition} sobre RFW/BFW é \emph{downstream}.}

\newword{Pixel information}{Quantidade de informação facial útil na imagem, tipicamente medida pela fração da imagem ocupada pela face após alinhamento e \emph{crop}. Identificada por Pangelinan et al. (2023) como possível \emph{driver} estrutural da disparidade em \emph{face recognition}.}

\newword{Confounder}{Variável que se correlaciona simultaneamente com a variável independente e com a variável dependente, podendo produzir associações espúrias caso não seja controlada. Nesta dissertação, \emph{pixel information} é tratada como \emph{confounder} na Etapa 5 do pipeline.}

\newword{Sensitivity analysis}{Procedimento metodológico em que os principais resultados são recomputados sob variações controladas de premissas críticas, testando a robustez das conclusões. Aplicada ao SkinToneNet no OE2.}

% 8. Conceitos de fairness, sociologia e legal
% ============================================================

\newword{Fairness}{Termo genérico para o conjunto de propriedades formais e informais que buscam garantir tratamento equitativo entre grupos demográficos em sistemas algorítmicos.}

\newword{Viés (bias)}{Discrepância sistemática de desempenho entre grupos demográficos que reflete padrões de treinamento, coleta de dados ou desenho arquitetural, e não diferenças intrínsecas da tarefa.}

\newword{Disparidade demográfica}{Variação mensurável de desempenho de um modelo entre grupos definidos por atributos demográficos (raça, gênero, idade, tom de pele).}

\newword{Pigmentocracia}{Sistema social de estratificação baseado no tom de pele. Documentado empiricamente em contextos latino-americanos pelo projeto PERLA (Telles 2014).}

\newword{Heterogeneidade fenotípica intra-categorial}{Diversidade fenotípica interna substantiva a uma mesma categoria racial nominal, fenômeno documentado empiricamente para a categoria Latinx/Hispanic e sustentado por evidência convergente de antropologia biológica, genética populacional e sociologia identitária.}

\newword{Latinx}{Termo neutro de gênero utilizado nesta dissertação para se referir à população identificada como Hispanic/Latino, seguindo padrão da literatura recente em \emph{fairness} e sociologia.}

\newword{European AI Act}{Regulamentação europeia publicada em 2024 que estabelece obrigatoriedade de auditoria de \emph{fairness} em sistemas de IA classificados como de alto risco, incluindo biometria facial.}

% 9. Instituições, normas e documentos regulatórios
% ============================================================

\newword{NIST}{National Institute of Standards and Technology --- agência do governo dos Estados Unidos responsável por padronização técnica. Conduziu, em 2019, a maior auditoria pública de algoritmos comerciais de \emph{face recognition}.}

\newword{NISTIR 8280}{Relatório técnico do NIST publicado em 2019, documentando diferenciais de dez a cem vezes na taxa de falso positivo entre grupos raciais em cento e oitenta e nove algoritmos comerciais.}

\newword{FRVT}{Face Recognition Vendor Test --- programa de avaliação sistemática do NIST sobre algoritmos comerciais de reconhecimento facial.}

\newword{CEP}{Comitê de Ética em Pesquisa --- órgão institucional responsável pela apreciação ética de projetos de pesquisa envolvendo seres humanos.}

\newword{Resolução 200/2021/CONSU}{Resolução do Conselho Universitário da Universidade Federal de São Paulo (Unifesp) que dispõe sobre diretrizes e procedimentos para projetos de pesquisa no CEP e na Comissão de Ética no Uso de Animais.}

\newword{Art. 8}{Artigo 8\textsuperscript{o} da Resolução 200/2021/CONSU --- dispensa de cadastro no CEP para projetos que não envolvem, direta ou indiretamente, seres humanos, nem animais vertebrados vivos. Aplica-se à presente dissertação.}

\newword{Declaração de Responsabilidade}{Documento exigido pelo parágrafo único do Art.~8\textsuperscript{o} da Resolução 200/2021/CONSU, assinado por estudante, orientador e chefe do Departamento, atestando o enquadramento do projeto na dispensa.}

\newword{AAPA Statement}{American Association of Physical Anthropologists Statement on Race and Racism (Fuentes et al. 2019) --- posição institucional que consolida o entendimento de que a categoria racial não representa acuradamente a variação biológica humana.}

\newword{PERLA}{Project on Ethnicity and Race in Latin America --- projeto de pesquisa multinacional coordenado por Edward Telles que documentou empiricamente pigmentocracia em Brasil, Colômbia, México e Peru.}
